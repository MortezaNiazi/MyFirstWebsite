j = int(input('shomare sotoon (andis g): '))
m = int(input('tedad adad ha (tedad adad dar har sotoon): '))
w_count = int(input('tedad tabaghat(tedad w ha dar har radif): '))


list_sotoon = []
print(f"adad haye liste sotoon ra vared konid (tedad: {m}):")
for _ in range(m):
    adad = int(input())
    list_sotoon.append(adad)


list_sotoon.sort()


maximum = max(list_sotoon)
minimum = min(list_sotoon)
L = (maximum - minimum) / w_count


t_values = [minimum + i * L for i in range(w_count)]
t_values.append(maximum)  


print(f"u({list_sotoon[0]})= 0")

for z in list_sotoon[1:]:
    u = []
    remaining_value = z - minimum

    for i in range(w_count):
        lower_bound = t_values[i]
        upper_bound = t_values[i + 1]

        if remaining_value <= 0:
            break

        if remaining_value <= (upper_bound - lower_bound):
            weight = remaining_value / (upper_bound - lower_bound)
            u.append(f"{weight:.2f}w{j}{i + 1}")
            remaining_value = 0
        else:
            weight = 1.0
            u.append(f"{weight:.2f}w{j}{i + 1}")
            remaining_value -= (upper_bound - lower_bound)

    print(f"u({z})= {' + '.join(u)}")
